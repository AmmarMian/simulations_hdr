def hex_to_rgba(h: str, a: float) -> str:
    """Convert a hex colour string to an rgba(...) CSS string."""
    h = h.lstrip("#")
    r, g, b = int(h[0:2], 16), int(h[2:4], 16), int(h[4:6], 16)
    return f"rgba({r},{g},{b},{a})"
